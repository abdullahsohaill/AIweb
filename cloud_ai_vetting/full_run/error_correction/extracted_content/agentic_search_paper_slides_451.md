\section{What is Agentic Search}
\begin{frame}{Definition}
  \begin{itemize}
    \item Agentic search enables autonomous agents to plan and execute multi-step search workflows.
    \item It supports query decomposition, iterative search, information synthesis, and reasoning.
    \item Unlike static search, agentic search allows dynamic interaction with information sources.
  \end{itemize}
\end{frame}

\section{Training-based}
